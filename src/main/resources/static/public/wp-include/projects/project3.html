const int analogInPin=A0;//电位器的输入引脚 int Value=0,Value1=0; //用于判断类比电压是否改变 float voltage=0,ref=4.68;//计算电压 void setup() { Serial.begin(9600); } void loop() { Value1=analogRead(analogInPin); if(Value1!=Value)//判断类比电压是否改变 { Value=Value1; voltage=Value*ref/1023.0;//电压计算 Serial.print("senor = "); Serial.print(Value); Serial.print(" voltage = "); Serial.print(voltage); Serial.println("V"); } delay(200); }



利用序列埠監控視窗顯示電位計的狀態-2
傳送到串列埠監控視窗中的數據，會在資料有變動時才傳到監控視窗中顯示，由於單位是mV，所以監控視窗中的資料變動率很高。
為了降低資料變動率，以提昇資料在監控視窗中的顯示穩定性，我們可以設定資料變動超過10mV時，才會將新的數值傳給串列埠監控視窗。我們只要將程式碼P3-3.ino的
if (Value1!=Value)
改成：
if ((Value1>=(Value+Value*0.01))||(Value1<=(Value-Value*0.01)))
即可。
const int analogInPin=A0;//电位器的输入引脚 int Value=0,Value1=0;//用于判断类比电压是否改变 float voltage=0,ref=4.68;//计算电压 void setup() { Serial.begin(9600); } void loop() { Value1=analogRead(analogInPin); //判断输入的类比电压是否有改变，若变动超过10mV显示新的电压值 if ((Value1>=(Value+Value*0.01))||(Value1<=(Value-Value*0.01))) { Value=Value1; voltage=Value*ref/1023.0;//电压计算 Serial.print("senor = "); Serial.print(Value); Serial.print(" voltage = "); Serial.print(voltage); Serial.println("V"); } delay(200); }
